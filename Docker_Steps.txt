*********Docker Steps*********


1. sudo apt update
2. sudo apt installdocker.io
3. sudo docker ps
4. sudo systemctl status docker 
5. sudo docker images
6. git clone <link>
7. cd <folder_name>
8. sudo docker build -t <image_name> .
9. sudo docker run --name <image_name> -d -p8000:8000 <image_name>
